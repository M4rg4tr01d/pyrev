Expand description

Query example:

use pyrev_ast::*;
use pyrev_ast_derive::*;
#[derive(Query, Debug)]
struct A {
    b: B,
    c: C,
}
#[derive(Query, Debug)]
struct B {
    d: String,
}
#[derive(Query, Debug, PartialEq)]
struct C {
    e: usize,
}
    let a = A {
        b: B {
            d: "hello".to_string(),
        },
        c: C {
            e: 1,
        },
    };
    let result = a.query::<C>();
    assert_eq!(
        result,
        vec![&C {
            e: 1,
        }]
    );
